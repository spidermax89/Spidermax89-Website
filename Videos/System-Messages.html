Local Script

local StarterGui = game:GetService("StarterGui")

local Messages = { -- The messages you would like to send!

"Put some messages here",

"Like the video",

"Sub if you enjoy this video",

"Test",

"Hello"

}

local TimeBetweenMessages = 1 -- Time Between each message (In Seconds!)

while wait(TimeBetweenMessages) do

StarterGui:SetCore("ChatMakeSystemMessage",{

Text = "[RoBot] " .. Messages[math.random(1,#Messages)],

Color = Color3.new(1, 0.215686, 0.921569), -- Default is Color3.fromRGB(255,255,243)

Font = Enum.Font.FredokaOne, -- Default is SourceSansBold

TextSize = 18 -- Default is 18 but can be bigger or smaller!

})

end
